const axios = require("axios"); axios.get("https://petstore.swagger.io/v2/pet/findByStatus?status=available") .then((res) => { const pets = res.data; const slicedpets = pets.slice(0, 10); for (pet of slicedpets) { console.log(`Dog id: ${pet.id}, Name: ${pet.name} `); } }) .catch((err) => { console.log("Error: ", err.message); });
